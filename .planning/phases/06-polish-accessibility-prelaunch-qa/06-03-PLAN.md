  ? `https://formspree.io/f/${process.env.NEXT_PUBLIC_FORMSPREE_ID}`
  : null

async function onSubmit(data: ContactFormData) {
  try {
    if (FORMSPREE_URL) {
      const res = await fetch(FORMSPREE_URL, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', Accept: 'application/json' },
        body: JSON.stringify(data),
      })
      if (!res.ok) {
        setSubmitError('Something went wrong. Please try again.')
        return
      }
    }
    setIsSubmitted(true)
  } catch {
    setSubmitError('Network error. Please try again.')
  }
}

**Changes to ContactSection.tsx:**

1. Add a `submitError` state: `const [submitError, setSubmitError] = useState<string | null>(null)`

2. Define the Formspree endpoint URL at module level (outside component):
```typescript
const FORMSPREE_URL = process.env.NEXT_PUBLIC_FORMSPREE_ID
  ? `https://formspree.io/f/${process.env.NEXT_PUBLIC_FORMSPREE_ID}`
  : null
```

3. Make `onSubmit` async and add Formspree fetch:
```typescript
async function onSubmit(data: ContactFormData) {
  setSubmitError(null)
  setIsSubmitting(true)
  try {
    if (FORMSPREE_URL) {
      const res = await fetch(FORMSPREE_URL, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json', Accept: 'application/json' },
        body: JSON.stringify(data),
      })
      if (!res.ok) {
        setSubmitError('Something went wrong. Please try again.')
        return
      }
    }
    setIsSubmitted(true)
  } catch {
    setSubmitError('Network error. Please check your connection and try again.')
  } finally {
    setIsSubmitting(false)
  }
}
```

4. Add a `submitting` state for loading feedback: `const [isSubmitting, setIsSubmitting] = useState(false)`. Disable the submit button and show "SENDING..." text while submitting:
```tsx
<button
  type="submit"
  disabled={isSubmitting}
  className="w-full py-3 border border-gold text-gold font-body text-sm tracking-widest uppercase transition-all duration-300 hover:bg-gold hover:text-void btn-glow focus-gold disabled:opacity-50 disabled:cursor-not-allowed"
>
  {isSubmitting ? 'SENDING...' : 'SEND MESSAGE'}
</button>
```

5. Display the error message below the submit button:
```tsx
{submitError && (
  <p className="mt-2 text-sm text-red-400 text-center" role="alert">
    {submitError}
  </p>
)}
```

6. Update the form's onSubmit to use handleSubmit with the async onSubmit. react-hook-form's handleSubmit already handles async functions.

**Graceful degradation:** When NEXT_PUBLIC_FORMSPREE_ID is not set, `FORMSPREE_URL` is null. The `if (FORMSPREE_URL)` check means the form just shows the success state as before -- no breaking change for development. This matches the existing v1 behavior.

**Do NOT install @formspree/react package.** The fetch-based approach works with the existing react-hook-form setup without adding a new dependency.
